-- ============================================
-- PROCEDIMIENTO: sp_ActualizarTurno
-- ============================================
CREATE   PROCEDURE [dbo].[sp_ActualizarTurno]
    @id_turno INT,
    @id_medico INT,
    @fecha DATE = NULL,
    @hora_inicio TIME(7) = NULL,
    @hora_fin TIME(7) = NULL,
    @estado VARCHAR(15) = NULL
AS
BEGIN
    SET NOCOUNT ON;
    
    -- Verificar que el turno existe y pertenece al médico
    IF NOT EXISTS (
        SELECT 1 FROM turno 
        WHERE id_turno = @id_turno AND id_medico = @id_medico
    )
    BEGIN
        RAISERROR('Turno no encontrado o no te pertenece', 16, 1);
        RETURN;
    END
    
    -- Actualizar solo los campos proporcionados
    UPDATE turno 
    SET 
        fecha = ISNULL(@fecha, fecha),
        hora_inicio = ISNULL(@hora_inicio, hora_inicio),
        hora_fin = ISNULL(@hora_fin, hora_fin),
        estado = ISNULL(@estado, estado)
    WHERE id_turno = @id_turno;
    
    SELECT 'Turno actualizado exitosamente' AS Mensaje;
END

-- ============================================
-- PROCEDIMIENTO: sp_CancelarTurno
-- ============================================
CREATE   PROCEDURE [dbo].[sp_CancelarTurno]
    @id_turno INT,
    @id_medico INT
AS
BEGIN
    SET NOCOUNT ON;
    
    -- Verificar que el turno existe y pertenece al médico
    IF NOT EXISTS (
        SELECT 1 FROM turno 
        WHERE id_turno = @id_turno AND id_medico = @id_medico
    )
    BEGIN
        RAISERROR('Turno no encontrado o no te pertenece', 16, 1);
        RETURN;
    END
    
    -- Verificar que el turno no esté ya cancelado o completado
    DECLARE @estado_actual VARCHAR(15);
    SELECT @estado_actual = estado FROM turno WHERE id_turno = @id_turno;
    
    IF @estado_actual IN ('cancelado', 'completado')
    BEGIN
        RAISERROR('El turno ya está %s', 16, 1, @estado_actual);
        RETURN;
    END
    
    -- Cancelar el turno
    UPDATE turno 
    SET estado = 'cancelado'
    WHERE id_turno = @id_turno;
    
    SELECT 'Turno cancelado exitosamente' AS Mensaje;
END

-- ============================================
-- PROCEDIMIENTO: sp_CrearTurno
-- ============================================
CREATE   PROCEDURE [dbo].[sp_CrearTurno]
    @fecha DATE,
    @hora_inicio TIME(7),
    @hora_fin TIME(7),
    @estado VARCHAR(15),
    @id_medico INT
AS
BEGIN
    SET NOCOUNT ON;
    
    -- Verificar que el médico no tenga conflicto de horario
    IF EXISTS (
        SELECT 1 FROM turno 
        WHERE id_medico = @id_medico 
        AND fecha = @fecha
        AND estado != 'cancelado'
        AND (
            (hora_inicio < @hora_fin AND hora_fin > @hora_inicio)
        )
    )
    BEGIN
        RAISERROR('Ya existe un turno en ese horario', 16, 1);
        RETURN;
    END
    
    -- Insertar el turno
    INSERT INTO turno (fecha, hora_inicio, hora_fin, estado, id_medico)
    VALUES (@fecha, @hora_inicio, @hora_fin, @estado, @id_medico);
    
    SELECT SCOPE_IDENTITY() AS id_turno;
END

-- ============================================
-- PROCEDIMIENTO: sp_ListarTurnosDisponibles
-- ============================================
CREATE   PROCEDURE [dbo].[sp_ListarTurnosDisponibles]
AS
BEGIN
    SELECT 
        t.id_turno, 
        t.fecha, 
        t.hora_inicio, 
        t.hora_fin, 
        t.estado,
        m.especialidad, 
        u.nombre AS medico_nombre, 
        u.email AS medico_email,
        u.telefono AS medico_telefono
    FROM turno t
    INNER JOIN medico m ON t.id_medico = m.id_medico
    INNER JOIN usuario u ON m.id_usuario = u.id_usuario
    WHERE t.estado = 'disponible'
    ORDER BY t.fecha ASC, t.hora_inicio ASC
END

-- ============================================
-- PROCEDIMIENTO: sp_ListarTurnosFiltrados
-- ============================================
CREATE   PROCEDURE [dbo].[sp_ListarTurnosFiltrados]
    @especialidad VARCHAR(100) = NULL,
    @fecha DATE = NULL,
    @estado VARCHAR(15) = NULL,
    @ubicacion VARCHAR(100) = NULL
AS
BEGIN
    SELECT 
        t.id_turno, 
        t.fecha, 
        t.hora_inicio, 
        t.hora_fin, 
        t.estado,
        m.especialidad, 
        u.nombre AS medico_nombre, 
        u.email AS medico_email
    FROM turno t
    INNER JOIN medico m ON t.id_medico = m.id_medico
    INNER JOIN usuario u ON m.id_usuario = u.id_usuario
    WHERE 1=1
        AND (@especialidad IS NULL OR m.especialidad LIKE '%' + @especialidad + '%')
        AND (@fecha IS NULL OR t.fecha = @fecha)
        AND (@estado IS NULL OR t.estado = @estado)
    ORDER BY t.fecha ASC, t.hora_inicio ASC
END

        pu.nombre AS paciente_nombre, 
        pu.email AS paciente_email, 
        pu.telefono AS paciente_telefono,
        pa.alergias
    FROM turno t
    LEFT JOIN inscripcion i ON t.id_turno = i.id_turno
    LEFT JOIN paciente p ON i.id_paciente = p.id_paciente
    LEFT JOIN usuario pu ON p.id_usuario = pu.id_usuario
    LEFT JOIN paciente pa ON p.id_paciente = pa.id_paciente
    WHERE t.id_medico = @id_medico
    ORDER BY t.fecha DESC, t.hora_inicio ASC
END

-- ============================================
-- PROCEDIMIENTO: sp_VerificarConflictoHorario
-- ============================================
CREATE   PROCEDURE [dbo].[sp_VerificarConflictoHorario]
    @id_medico INT,
    @fecha DATE,
    @hora_inicio TIME(7),
    @hora_fin TIME(7),
    @excluir_turno INT = NULL
AS
BEGIN
    SELECT COUNT(*) AS conflicto
    FROM turno 
    WHERE id_medico = @id_medico 
        AND fecha = @fecha
        AND estado != 'cancelado'
        AND (@excluir_turno IS NULL OR id_turno != @excluir_turno)
        AND (
            (hora_inicio < @hora_fin AND hora_fin > @hora_inicio)
        )
END
