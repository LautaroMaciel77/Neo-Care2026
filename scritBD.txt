-- ============================================
-- PROCEDIMIENTO: sp_ActualizarTurno
-- ============================================
CREATE PROCEDURE [dbo].[sp_ActualizarTurno]
    @id_turno INT,
    @id_medico INT,
    @fecha DATE = NULL,
    @hora_inicio TIME(7) = NULL,
    @hora_fin TIME(7) = NULL,
    @estado VARCHAR(15) = NULL
AS
BEGIN
    SET NOCOUNT ON;
    
    IF NOT EXISTS (
        SELECT 1 FROM turno 
        WHERE id_turno = @id_turno AND id_medico = @id_medico
    )
    BEGIN
        RAISERROR('Turno no encontrado o no te pertenece', 16, 1);
        RETURN;
    END
    
    UPDATE turno 
    SET 
        fecha = ISNULL(@fecha, fecha),
        hora_inicio = ISNULL(@hora_inicio, hora_inicio),
        hora_fin = ISNULL(@hora_fin, hora_fin),
        estado = ISNULL(@estado, estado)
    WHERE id_turno = @id_turno;
    
    SELECT 'Turno actualizado exitosamente' AS Mensaje;
END

GO

-- ============================================
-- PROCEDIMIENTO: sp_CancelarTurno
-- ============================================
CREATE PROCEDURE [dbo].[sp_CancelarTurno]
    @id_turno INT,
    @id_medico INT
AS
BEGIN
    SET NOCOUNT ON;
    
    IF NOT EXISTS (
        SELECT 1 FROM turno 
        WHERE id_turno = @id_turno AND id_medico = @id_medico
    )
    BEGIN
        RAISERROR('Turno no encontrado o no te pertenece', 16, 1);
        RETURN;
    END
    
    DECLARE @estado_actual VARCHAR(15);
    SELECT @estado_actual = estado FROM turno WHERE id_turno = @id_turno;
    
    IF @estado_actual IN ('cancelado', 'completado')
    BEGIN
        RAISERROR('El turno ya está %s', 16, 1, @estado_actual);
        RETURN;
    END
    
    UPDATE turno SET estado = 'cancelado' WHERE id_turno = @id_turno;
    
    -- También cancelar la inscripción si existe
    UPDATE inscripcion SET fecha_inscripcion = NULL WHERE id_turno = @id_turno;
    
    SELECT 'Turno cancelado exitosamente' AS Mensaje;
END

-- ============================================
-- PROCEDIMIENTO: sp_CrearTurno
-- ============================================
CREATE PROCEDURE [dbo].[sp_CrearTurno]
    @fecha DATE,
    @hora_inicio TIME(7),
    @hora_fin TIME(7),
    @estado VARCHAR(15),
    @id_medico INT
AS
BEGIN
    SET NOCOUNT ON;
    
    IF EXISTS (
        SELECT 1 FROM turno 
        WHERE id_medico = @id_medico 
        AND fecha = @fecha
        AND estado != 'cancelado'
        AND ((hora_inicio < @hora_fin AND hora_fin > @hora_inicio))
    )
    BEGIN
        RAISERROR('Ya existe un turno en ese horario', 16, 1);
        RETURN;
    END
    
    INSERT INTO turno (fecha, hora_inicio, hora_fin, estado, id_medico)
    VALUES (@fecha, @hora_inicio, @hora_fin, @estado, @id_medico);
    
    SELECT SCOPE_IDENTITY() AS id_turno;
END

CREATE PROCEDURE [dbo].[sp_ListarTurnosFiltrados]
    @especialidad VARCHAR(100) = NULL,
    @fecha DATE = NULL,
    @estado VARCHAR(15) = NULL,
    @id_localidad INT = NULL
AS
BEGIN
    SELECT 
        t.id_turno, 
        t.fecha, 
        t.hora_inicio, 
        t.hora_fin, 
        t.estado,
        m.especialidad, 
        u.nombre AS medico_nombre, 
        u.email AS medico_email,
        l.nombre_localidad,
        p.nombre_provincia
    FROM turno t
    INNER JOIN medico m ON t.id_medico = m.id_medico
    INNER JOIN usuario u ON m.id_usuario = u.id_usuario
    LEFT JOIN localidad l ON u.id_localidad = l.id_localidad
    LEFT JOIN provincia p ON l.id_provincia = p.id_provincia
    WHERE 1=1
        AND (@especialidad IS NULL OR m.especialidad LIKE '%' + @especialidad + '%')

        AND (@estado IS NULL OR t.estado = @estado)
        AND (@id_localidad IS NULL OR u.id_localidad = @id_localidad)
    ORDER BY t.fecha ASC, t.hora_inicio ASC
END

CREATE   PROCEDURE [dbo].[sp_ModificarHorarioTurno]
    @id_turno INT,
    @id_medico INT,
    @nueva_fecha DATE,
    @nueva_hora_inicio TIME(7),
    @nueva_hora_fin TIME(7)
AS
BEGIN
    SET NOCOUNT ON;
    
    DECLARE @tiene_paciente BIT = 0;
    DECLARE @nombre_paciente VARCHAR(100) = NULL;
    DECLARE @email_paciente VARCHAR(100) = NULL;
    
    -- Verificar que el turno existe y pertenece al médico
    IF NOT EXISTS (
        SELECT 1 FROM turno 
        WHERE id_turno = @id_turno AND id_medico = @id_medico
    )
    BEGIN
        RAISERROR('Turno no encontrado o no te pertenece', 16, 1);
        RETURN;
    END
    
    -- Verificar si tiene paciente inscrito
    SELECT 
        @tiene_paciente = 1,
        @nombre_paciente = u.nombre,
        @email_paciente = u.email
    FROM inscripcion i
    INNER JOIN paciente p ON i.id_paciente = p.id_paciente
    INNER JOIN usuario u ON p.id_usuario = u.id_usuario
    WHERE i.id_turno = @id_turno;
    
    -- Verificar conflicto de horario (excluyendo este turno)
    IF EXISTS (
        SELECT 1 FROM turno 
        WHERE id_medico = @id_medico 
            AND id_turno != @id_turno
            AND fecha = @nueva_fecha
            AND estado != 'cancelado'
            AND (
                (hora_inicio < @nueva_hora_fin AND hora_fin > @nueva_hora_inicio)
            )
    )
    BEGIN
        RAISERROR('Ya existe otro turno en ese horario', 16, 1);
        RETURN;
    END
    
    -- Actualizar el turno
    UPDATE turno 
    SET 
        fecha = @nueva_fecha,
        hora_inicio = @nueva_hora_inicio,
        hora_fin = @nueva_hora_fin
    WHERE id_turno = @id_turno;
    
    -- Devolver información
    SELECT 
        'Horario modificado exitosamente' AS Mensaje,
        @tiene_paciente AS TienePaciente,
        @nombre_paciente AS NombrePaciente,
        @email_paciente AS EmailPaciente;
END

-- ============================================
-- PROCEDIMIENTO: sp_RegistrarAtencionMedica
-- ============================================
CREATE   PROCEDURE [dbo].[sp_RegistrarAtencionMedica]
    @id_inscripcion INT,
    @sintomas VARCHAR(500) = NULL,
    @diagnostico VARCHAR(255) = NULL,
    @tratamiento VARCHAR(500) = NULL,
    @receta VARCHAR(500) = NULL,
    @notas VARCHAR(500) = NULL,
    @fecha_atencion DATETIME = NULL
AS
BEGIN
    SET NOCOUNT ON;
    
    -- Verificar que la inscripción existe
    IF NOT EXISTS (SELECT 1 FROM inscripcion WHERE id_inscripcion = @id_inscripcion)
    BEGIN
        RAISERROR('La inscripción no existe', 16, 1);
        RETURN;
    END
    
    -- Verificar que no haya una historia médica ya registrada para esta inscripción
    IF EXISTS (SELECT 1 FROM historia_medica WHERE id_inscripcion = @id_inscripcion)
    BEGIN
        RAISERROR('Ya existe una historia médica para esta inscripción', 16, 1);
        RETURN;
    END
    
    -- Insertar la historia médica
    INSERT INTO historia_medica (
        id_inscripcion, 
        sintomas, 
        diagnostico, 
        tratamiento, 
        receta, 
        notas, 
        fecha_atencion
    )
    VALUES (
        @id_inscripcion,
        @sintomas,
        @diagnostico,
        @tratamiento,
        @receta,
        @notas,
        ISNULL(@fecha_atencion, GETDATE())
    );
    
    -- Actualizar el estado del turno a 'completado'
    UPDATE t 
    SET t.estado = 'completado'
    FROM turno t
    INNER JOIN inscripcion i ON t.id_turno = i.id_turno
    WHERE i.id_inscripcion = @id_inscripcion;
    
    SELECT SCOPE_IDENTITY() AS id_historia_medica;
END

CREATE PROCEDURE [dbo].[sp_VerificarConflictoHorario]
    @id_medico INT,
    @fecha DATE,
    @hora_inicio TIME(7),
    @hora_fin TIME(7),
    @excluir_turno INT = NULL
AS
BEGIN
    SELECT COUNT(*) AS conflicto
    FROM turno 
    WHERE id_medico = @id_medico 
        AND fecha = @fecha
        AND estado != 'cancelado'
        AND (@excluir_turno IS NULL OR id_turno != @excluir_turno)
        AND ((hora_inicio < @hora_fin AND hora_fin > @hora_inicio))
END
